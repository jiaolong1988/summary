
select t.bizmetadata1, t.pkuuid, t.archdate
  from KS06_XDB_METADATA t
 group by t.bizmetadata1, t.pkuuid, t.archdate
   
 
                                                                         
select t1.*, rowid
  from KS06_XDB_METADATA t1
 where t1.rowid in (select min(rowid)
                      from KS06_XDB_METADATA t2
                     where t1.bizmetadata1 = t2.bizmetadata1
                       and t1.pkuuid = t2.pkuuid
                       and t1.archdate = t2.archdate);
