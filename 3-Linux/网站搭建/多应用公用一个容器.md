```bash
 #创建自定义网络（名称自定义，如 app-network）
docker network create app-network
```

```bash
docker run -d \
  --name mysql-server \  # MySQL 容器名称（后续应用连接用）
  --network app-network \  # 加入自定义网络
  -e MYSQL_ROOT_PASSWORD=root@123456 \  # root 密码（必须设置，自定义强密码）
  -v mysql-data:/var/lib/mysql \  # 数据卷持久化（关键！防止数据丢失）
  --restart=always \  # 容器退出自动重启
  mysql:8.0  # 镜像版本（8.0 稳定，兼容大部分应用）
  --max_allowed_packet=64M
  
  #密码是：root@123456
  
docker run -d \
  -p 3305:3306 \
  --name mysql-server \
  --network app-network \
  -e MYSQL_ROOT_PASSWORD=root@123456 \
  -v mysql-data:/var/lib/mysql \
  --restart=always \
  mysql:5.7
```

```
# 进入 MySQL 容器的交互式终端（使用 root 登录）
docker exec -it mysql-server mysql -u root -p

-- 1. 为 WordPress 创建数据库和用户
CREATE DATABASE wordpress_db CHARACTER SET utf8mb4 COLLATE utf8mb4_unicode_ci;
CREATE USER 'wp_user'@'%' IDENTIFIED BY 'wp_pass@123';  -- '%' 允许所有网络访问（仅Docker网络内可用）
GRANT ALL PRIVILEGES ON wordpress_db.* TO 'wp_user'@'%';  -- 仅授权 wordpress_db 数据库

-- 2. 为 TestApp 创建数据库和用户
CREATE DATABASE testapp_db CHARACTER SET utf8mb4 COLLATE utf8mb4_unicode_ci;
CREATE USER 'testapp_user'@'%' IDENTIFIED BY 'testapp_pass@123';
GRANT ALL PRIVILEGES ON testapp_db.* TO 'testapp_user'@'%';

-- 刷新权限生效
FLUSH PRIVILEGES;

-- 退出 MySQL 终端
EXIT;


CREATE DATABASE wordpress_db CHARACTER SET utf8mb4 COLLATE utf8mb4_unicode_ci;
CREATE USER 'wp_user'@'%' IDENTIFIED BY 'wp_pass@123';  
GRANT ALL PRIVILEGES ON wordpress_db.* TO 'wp_user'@'%';
```

```
docker run -d \
  --name wordpress \
  --network app-network \  # 加入同一自定义网络
  -p 80:80 \  # 端口映射（自定义，避免冲突）
  -e WORDPRESS_DB_HOST=mysql-server \  # MySQL 容器名称（关键！）
  -e WORDPRESS_DB_NAME=wordpress_db \  # 专属数据库
  -e WORDPRESS_DB_USER=wp_user \  # 专属用户
  -e WORDPRESS_DB_PASSWORD=wp_pass@123 \  # 专属密码
  -v wp-data:/var/www/html \  # 持久化 WordPress 数据
  --restart=always \
  wordpress:latest
  
  
docker run -d \
--name wordpress \
--network app-network \
-p 8888:80 \
-e WORDPRESS_DB_HOST=mysql-server \

-v /root/wordpress-New/wordpress_data:/var/www/html \
-v /root/wordpress-New/custom-php.ini:/usr/local/etc/php/conf.d/custom.ini \
--restart=always \
wordpress:latest
```
