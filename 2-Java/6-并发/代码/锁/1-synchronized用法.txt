class X {
  // 修饰非静态方法
  synchronized void foo() {
    // 临界区
  }
  // 修饰静态方法
  synchronized static void bar() {
    // 临界区
  }
  // 修饰代码块
  Object obj = new Object()；
  void baz() {
    synchronized(obj) {
      // 临界区
    }
  }
}  


//当修饰静态方法的时候，锁定的是当前类的 Class 对象，在上面的例子中就是 Class X；
//当修饰非静态方法的时候，锁定的是当前实例对象 this。
